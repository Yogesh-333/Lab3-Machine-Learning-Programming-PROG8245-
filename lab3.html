    name = fake.name()

# Step 1: Create a projects table
cur = conn.cursor()
cur.execute("""
DROP TABLE IF EXISTS projects CASCADE;
CREATE TABLE projects (
    project_id SERIAL PRIMARY KEY,
    project_name VARCHAR(100),
    deadline DATE,
    priority VARCHAR(20)
);
""")

# Step 2: Insert some sample projects
project_names = ["IBM", "Asgard", "Venilla_v1", "Saturn_v2", "Zephyr"]
priorities = ["High", "Medium", "Low"]

for name in project_names:
    deadline = fake.date_between(start_date=dt.date(2024, 6, 1), end_date=dt.date(2025, 12, 31))
    priority = random.choice(priorities)
    cur.execute("INSERT INTO projects (project_name, deadline, priority) VALUES (%s, %s, %s)", (name, deadline, priority))

conn.commit()

# Step 3: Create project_assignment table (many-to-many relationship)
cur.execute("""
DROP TABLE IF EXISTS project_assignment;
CREATE TABLE project_assignment (
    assignment_id SERIAL PRIMARY KEY,
    employee_id INTEGER REFERENCES employees(employee_id),
    project_id INTEGER REFERENCES projects(project_id)
);
""")

# Step 4: Assign each employee to 1–3 random projects
cur.execute("SELECT employee_id FROM employees;")
employee_ids = [row[0] for row in cur.fetchall()]

cur.execute("SELECT project_id FROM projects;")
project_ids = [row[0] for row in cur.fetchall()]

assignments = []
for emp_id in employee_ids:
    assigned_projects = random.sample(project_ids, random.randint(1, 3))
    for proj_id in assigned_projects:
        assignments.append((emp_id, proj_id))

for a in assignments:
    cur.execute("INSERT INTO project_assignment (employee_id, project_id) VALUES (%s, %s);", a)

conn.commit()


# Step 5: Query to join all tables
df_proj = pd.read_sql("""
SELECT

JOIN departments d ON e.department_id = d.department_id

8 Insights Analysis¶

1. Average Salary by Position and Start Year (Bar Chart)¶

This bar chart showcases trends in average salaries across different job positions over various start years:

  • Cloud Architect consistently commands high salaries across nearly all years, especially in 2020, 2022, and 2024, indicating strong demand.
  • Software Engineers also show substantial salary peaks, notably in 2015, 2018, and 2024, suggesting periodic market surges.
  • DevOps Engineers see a notable spike in 2020, likely reflecting increased demand for CI/CD and infrastructure automation during the remote work boom.
  • QA Testers and IT Managers display fluctuating salary trends, indicating variability in demand or budget allocation.
  • Security Specialists maintain moderate but consistent salaries, with smaller fluctuations compared to other roles.
  • Data Analyst data appears less frequent but shows competitive pay when present, such as in 2016 and 2024.

Key Insight: Technical leadership roles like Cloud Architect and high-demand development roles like Software Engineer and DevOps Engineer generally command higher salaries. Spikes in 2020 and 2024 may align with broader digital transformation and security needs.

2. Average Salary by Project and Department (Heatmap)¶

This heatmap visualizes average salaries across departments for various projects:

  • Security departments consistently offer the highest salaries, with Project Zephyr hitting the peak at $167,713.
  • IT Operations departments also offer strong compensation, particularly in Asgard and Saturn_v2, with salaries above $140,000.
  • Data Science salaries are relatively strong, especially in Zephyr ($132,748) and **Saturn_v2** ($121,569).
  • Engineering tends to offer the lowest average salaries, particularly in Asgard and Vanilla_v1.
  • Product departments show moderate salary levels across all projects.

Key Insight: Projects investing in Security and IT Operations tend to allocate more budget toward compensation, suggesting their critical roles in infrastructure and risk management. Zephyr is the most lucrative project overall.
